

-- 1) Create Database and use it
CREATE DATABASE UniversityDB;
USE UniversityDB;

-- 2) Create Department table (Unique constraint on dept_name)
CREATE TABLE Department (
    dept_no INT,
    dept_name VARCHAR(50) UNIQUE,
    building_name VARCHAR(50)
);

-- 3) Create Instructor table (NOT NULL constraint on ins_name)
CREATE TABLE Instructor (
    ins_id INT,
    ins_name VARCHAR(50) NOT NULL,
    dept_no INT,
    salary DECIMAL(10,2),
    mob_no VARCHAR(15)
);

-- 4) Create Course table
CREATE TABLE Course (
    course_id INT,
    title VARCHAR(50),
    dept_no INT,
    credits INT
);

-- 5) Create Teaches table
CREATE TABLE Teaches (
    teacher_id INT,
    course_id INT,
    semester VARCHAR(20),
    year INT
);

-- 6) Add PRIMARY KEY to Department table
ALTER TABLE Department
ADD CONSTRAINT pk_dept PRIMARY KEY (dept_no);

-- 7) Add FOREIGN KEY in Instructor table
ALTER TABLE Instructor
ADD CONSTRAINT fk_deptno FOREIGN KEY (dept_no) REFERENCES Department(dept_no);

-- 8) Modify Department table by adding column 'budget'
ALTER TABLE Department
ADD budget DECIMAL(12,2);

-- 9) Create UNIQUE INDEX on mobile number of Instructor table
CREATE UNIQUE INDEX idx_mobile ON Instructor(mob_no);

-- 10) Create a VIEW of Instructor table (excluding salary field)
CREATE VIEW instr_view AS
SELECT ins_id, ins_name, dept_no, mob_no FROM Instructor;

-- 15) Remove the 'budget' column from Department table
ALTER TABLE Department

RENAME TABLE Course TO Course_Details;

SELECT i.ins_name, c.title
FROM Instructor i
JOIN Teaches t ON i.ins_id = t.teacher_id
JOIN Course_Details c ON t.course_id = c.course_id;
